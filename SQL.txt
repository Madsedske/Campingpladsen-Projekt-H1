CREATE VIEW [dbo].[Rengøring af hytter]
AS
SELECT        dbo.CampingSite.SiteNumber, dbo.CampingSite.SiteType, dbo.Reservation.ReservationEndDate, dbo.Reservation.YesOrNoCleaning
FROM            dbo.Reservation INNER JOIN
                         dbo.CampingSite ON dbo.Reservation.SiteNumber = dbo.CampingSite.SiteNumber
WHERE        (dbo.Reservation.YesOrNoCleaning = 'Yes') AND (dbo.Reservation.ReservationEndDate = CONVERT(varchar(10), GETDATE(), 120))

CREATE VIEW [dbo].[Kunder der kommer i dag]
